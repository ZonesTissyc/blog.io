void preorder(Tree root) {
    if (!root) return;
    stack<Tree> s;
    s.push(root);
    while (!s.empty()) {
        Tree node = s.top(); s.pop();
        cout << node->data << ' ';
        if (node->RChild) s.push(node->RChild);
        if (node->LChild) s.push(node->LChild);
    }
}

void inorder(Tree root) {
    stack<Tree> s;
    Tree p = root;
    while (p || !s.empty()) {
        if (p) {
            s.push(p);
            p = p->LChild;
        } else {
            p = s.top(); s.pop();
            cout << p->data << ' ';
            p = p->RChild;
        }
    }
}

void postorder(Tree root) {
    if (!root) return;
    stack<Tree> s1, s2;
    s1.push(root);
    while (!s1.empty()) {
        Tree node = s1.top(); s1.pop();
        s2.push(node);
        if (node->LChild) s1.push(node->LChild);
        if (node->RChild) s1.push(node->RChild);
    }
    while (!s2.empty()) {
        cout << s2.top()->data << ' ';
        s2.pop();
    }
}
